#science ORF
science=read.xlsx("data/science.adk3487_tables_s1_to_s14 2.xlsx",sheet = "ST6_ncHLAp PDAC PDO  ",startRow = 2)
sub_sci=unique(science[,c("seq_upper","filename","biotype","hh_ip_detection","ribo_detection","thymus_ip_detection","cancer_restricted")])

#remove hh,ribo,thymu
sub_sci=sub_sci[sub_sci$hh_ip_detection==F&sub_sci$ribo_detection==F&sub_sci$thymus_ip_detection==F,]

sub_sci_uni=unique(sub_sci[,-2])

#uniform define
sci_type=as.data.frame(table(sub_sci_uni$biotype))
sci_type$Region=as.character(sci_type$Var1)
sci_type$Region[sci_type$Region=="Antisense"|sci_type$Region=="lincRNA"|sci_type$Region=="ncRNA Processed Transcript"|sci_type$Region=="ncRNA Retained Intron"|sci_type$Region=="Pseudogene"|sci_type$Region=="TUCP"]="ncRNA"
sci_type$Region[sci_type$Region=="Nonsense Mediated Decay"]="LowConfidence\nExon"
sci_type$Region[sci_type$Region=="Out-of-Frame (intORF)"]="InternalORF"
sci_type$Region[sci_type$Region=="3' dORF"]="dORF"
sci_type$Region[sci_type$Region=="5' uORF"]="uORF"
sci_type=sci_type%>%group_by(Region)%>%summarise(Freq=sum(Freq))
sci_type=sci_type[order(sci_type$Freq),]
sci_type$Region=factor(sci_type$Region,levels = unique(sci_type$Region))
sci_type$freq=sci_type$Freq/sum(sci_type$Freq)
sci_type$label=paste(sci_type$Freq,"(",round(sci_type$freq,2)*100,"%)",sep="")

sci_p=ggplot(sci_type,aes(y=Region,x=Freq,fill=Region))+geom_col(alpha=0.8)+

  xlab("No. of ncHLAp")+ylab("")+
  scale_x_continuous(expand =expansion(mult = c(0, 0.27)))+
  ggtitle("Science(517)")
