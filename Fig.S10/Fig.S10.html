library(foreach)
library(reshape2)
library(dplyr)
library(venn)
fl=dir("data/annotation")[-c(2,3)]
dt=foreach(i=1:length(fl)) %do% read.csv(paste("data/annotation/",fl[i],sep=""))
neg_TRA=read.delim("data/annotation/Sample_JZ25214718-1023-01-2025BA44A_14R1_nofilter_4.7_SC.clones_TRA.tsv")
neg_TRB=read.delim("data/annotation/Sample_JZ25214718-1023-01-2025BA44A_14R1_nofilter_4.7_SC.clones_TRB.tsv")

df=list()
for(i in 1:length(fl)){
  stat=dcast(dt[[i]],barcode~chain)
  stat_filter=stat[stat$TRA==1&stat$TRB==1,]
  dt[[i]]=dt[[i]][is.element(dt[[i]]$barcode,stat_filter$barcode),]
  dt[[i]]=dt[[i]][dt[[i]]$chain=="TRA"|dt[[i]]$chain=="TRB",]
  dt[[i]]$cdr3_aa=paste(dt[[i]]$chain,dt[[i]]$cdr3,sep=":")
  
  cell_aa=dt[[i]][order(dt[[i]]$barcode,dt[[i]]$chain),]%>%group_by(barcode)%>%summarise(cdr3_pair=paste0(cdr3_aa,collapse = ";"))
  #write.csv(cell_aa,paste("res/",sub("_contig_annotations.csv","_contig_annotations_filteredcontig.csv",fl[i]),sep=""),row.names = F)
  
  df[[i]]=cell_aa%>%group_by(cdr3_pair)%>%summarise(frequency=length(barcode))
  df[[i]]$proportion=df[[i]]$frequency/sum(df[[i]]$frequency)
  df[[i]]$TRA=gsub("TRA:|;.*","",df[[i]]$cdr3_pair)
  df[[i]]$TRB=gsub(".*TRB:","",df[[i]]$cdr3_pair)
  # write.csv(df[[i]],paste("res/",sub("_contig_annotations.csv","_clonotypes.csv",fl[i]),sep=""),row.names = F)
  df[[i]]=df[[i]][!is.element(df[[i]]$TRA,neg_TRA$aaSeqCDR3),]
  df[[i]]=df[[i]][!is.element(df[[i]]$TRB,neg_TRB$aaSeqCDR3),]
  #write.csv(df[[i]],paste("res/",sub("_contig_annotations.csv","_clonotypes_filter44A.csv",fl[i]),sep=""),row.names = F)
}
